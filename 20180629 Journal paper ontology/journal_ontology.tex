	\node[tag, below of=taga, node distance=3em, xshift=#3](tagb#2){#1};
	\draw[tagarrow] (taga) -- (helper) -| (tagb#2);
	\node[coordinate, xshift=1em](tagb#2 helper) at (tagb#2.south west) {};
}
